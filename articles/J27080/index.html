STOP_ON_OLD_FILE=0

case "$1" in

-u) URL=$2; shift 2;;

-f) FOLDER=$2; shift 2;;

--) shift ; break ;;

esac

usage

fi

usage

usage

exit 1

fi

#

FILENAME=`echo $LINK | sed -r -e 's/.*\/([^\/]+)\?.*/\1/'`

if [ -e $FOLDER/$FILENAME ]; then

if [ "$DELETE_OLD_FILES" -eq "1" ]; then

rm $FOLDER/$FILENAME

wget -cq -O $FOLDER/$FILENAME $LINK

let DONE=$DONE+1

fi

if [ "$STOP_ON_OLD_FILE" -eq "1" ]; then

echo $DONE

exit 0

fi

else

wget -cq -O $FOLDER/$FILENAME $LINK

let DONE=$DONE+1

fi

done

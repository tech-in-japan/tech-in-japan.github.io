from os import walk

from os.path import join



def find(path, ext):

for dir, _, files in walk(path):

for f in files:

if f.endswith(ext):

yield join(dir, f)



#print '\n' .join(find( 'src' , '.java' ))



def go():

else :

#print "Obsolete: " , f

print 'will compile:' , to_compile

print 'obsolete:' , obsolete



out .close()

//4800 GPS

//8 , 2

lcd.begin(8, 2);

void loop() {

bool newdata = false ;

unsigned long start = millis();

long lat, lon;

unsigned long age;

//

while (millis() - start < 1000) {

if (readgps())

newdata = true ;

}

if (newdata) {

lcd.setCursor(0, 0);

lcd.print(lat);

lcd.setCursor(0, 1);

lcd.print(lon);

}

}

bool readgps() {

while (nss.available()) {

int b = nss.read();

// TinyGPS , \r \n

if ( '\r' != b) {

if (gps.encode(b))

return true ;

}

}

return false ;

}

Zoo.Dog = Class

({

    base: Zoo.Animal,



    construct: function(name, bark, loudness)

    {

        this.$base()(name);

        this._bark = bark;

        this._loudness = loudness;

    },



    methods:

    {

        Scream: function()

        {

            var s = "";

            for (var i = 0; i < this._loudness; i++)

            {

                s += this._bark;

            }

            document.write(s);

        },



        SayName: function()

        {

            document.write(this._bark + " ");

            this.$base('SayName')();

        }

    }

})

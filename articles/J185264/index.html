public static bool IsAdmin()
{
	System.Security.Principal.WindowsIdentity id = System.Security.Principal.WindowsIdentity.GetCurrent();

	return p.IsInRole(System.Security.Principal.WindowsBuiltInRole.Administrator);            
}

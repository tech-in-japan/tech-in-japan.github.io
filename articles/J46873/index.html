<script>jQuery.noConflict();</script> /* framework */

<script>

function moveTo(divId, mainId) {

jQuery( "#" +mainId).stop(); /* */

var topIn = jQuery( "#" +divId).css( "top" );

var leftIn = jQuery( "#" +divId).css( "left" );

leftIn = leftIn.replace(/px/, "" );

leftIn = leftIn - 400;

jQuery( "#" +mainId).animate({

top: "-" +topIn,

left: "-" +leftIn

}, 1500 );

return false ;

}

</script>

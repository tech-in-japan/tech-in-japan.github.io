def add_disk_check(hostname,zabbix)

'description'=>name,

'key_'=>"vfs.fs.size[#{disk},pfree]",

'hostid'=>hostid,

'type'=>'0',

'data_type'=>0,

'value_type'=>0,

'units'=>"%"

begin

'hostid' => hostid,

'expression'=>expression,

'description'=>description,

'priority'=>HIGH,

'status'=>ENABLE,

}

uid = zabbix.trigger.create(item_t)

p "cant create trigger on #{hostname} disk #{disk}" if uid.nil?

end



end

array(60 * 60 * 24 , 'D '),

array(60 * 60 , 'h:'),

array(60 , 'min'),

);

$today = time();

$since = $today - $original;

for ($i = 0, $j = count($chunks); $i < $j; $i++) {

$seconds = $chunks[$i][0];

$name = $chunks[$i][1];

if (($count = floor($since / $seconds)) != 0) {

break;

}

}

$print = "$count{$name}";

if ($i + 1 < $j) {

$seconds2 = $chunks[$i + 1][0];

$name2 = $chunks[$i + 1][1];

if (($count2 = floor(($since - ($seconds * $count)) / $seconds2)) != 0) {

$print .= "$count2{$name2}";

}

}

return $print;

}

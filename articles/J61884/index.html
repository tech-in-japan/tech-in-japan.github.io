int bfs( int s, int t)

{

init();

queue< int > Q;

mark[s] = 1;

pred[s] = s;

push[s] = MAX_VAL;



Q.push(s);

while ( !mark[t] && !Q.empty() )

{

int u = Q.front(); Q.pop();

for ( int v = 1; v < N; v++)

if ( !mark[v] && (C[u][v]-F[u][v] > 0) )

{

push[v] = min(push[u], C[u][v]-F[u][v]);

mark[v] = 1;

pred[v] = u;

Q.push(v);

}

}



return mark[t];

}

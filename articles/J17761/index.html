{

$xmlStr=str_replace('>','>',$xmlStr);

$xmlStr=str_replace('"','"',$xmlStr);

$xmlStr=str_replace("'",''',$xmlStr);

$xmlStr=str_replace("&",'&',$xmlStr);

return $xmlStr;

}

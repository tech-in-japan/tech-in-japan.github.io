function win2uni ( $s )

{

$s = str_replace ( '—ñ' , 'i' , $s ) ;

$s = str_replace ( '–Ü' , 'I' , $s ) ;

$s = convert_cyr_string ( $s , 'w' , 'i' ) ; // win1251 -> iso8859-5

// iso8859-5 -> unicode:

for ( $result = '' , $i = 0 ; $i < strlen ( $s ) ; $i ++ ) {

$charcode = ord ( $s [ $i ] ) ;

$result .= ( $charcode > 175 ) ? "&#" . ( 1040 + ( $charcode - 176 ) ) . ";" : $s [ $i ] ;

}

return $result ;

}

/**

* Run process in background with out buffer to file

* @param str $commandJob

* @param str $file_out_put

* @return int or false

*/

function PsExec($commandJob, $file_out_put) {

exec($command ,$op);

$pid = ( int )$op[0];

return false ;

}



}



return false ;

}

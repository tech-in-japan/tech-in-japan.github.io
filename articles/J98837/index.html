このトピックに触発されました 。 引っ越しして新しい居住地になり、緊急にインターネットが必要になりました。 199 UAHのKyivstarモデムの広告を見たので、このキットを購入しました。 NetworkManagerを介してUbuntuをセットアップするのに手間はかかりませんでした。 しかし、アカウントをチェックするのに十分なお金がありませんでした（ブラウザ経由ではありません）。 これを解決するのに役立つスクリプトを紹介します。







<?php



function decodePDU($ in ) {

$b = 0; $d = 0;

$ out = "" ;

foreach (str_split($ in , 2) as $ss) {

$ byte = hexdec($ss);

$c = (($ byte & ((1 << 7-$d)-1)) << $d) | $b;

$b = $ byte >> (7-$d);

$ out .= chr($c);

$d++;

if ($d == 7) {

$ out .= chr($b);

$d = 0; $b = 0;

}

}

return $ out ;

}



function encodePDU($ in ) {

$ out = "" ;

for ($i = 0; $i < strlen($ in ); $i++) {

$t = $i%8+1;

if ($t == 8)

continue ;

$c = ord($ in [$i])>>($i%8);

$oc = $c;

$b = ord($ in [$i+1]) & ((1 << $t)-1);

$c = ($b << (8-$t)) | $c;

$ out .= strtoupper(str_pad(dechex($c), 2, '0' , STR_PAD_LEFT));

}

return $ out ;

}

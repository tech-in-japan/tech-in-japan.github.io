function getContactList($content_page)

{

$result = array();



preg_match( '/"cla".*?D\((.+?)\);/is' , $content_page, $all);

if ( !empty($all[1]) )

{

preg_match_all( '/\["ce"(.+?)\]/' , $all[1], $list);



if ( !empty($list[1]) )

{

foreach ( $list[1] as $item )

{

if ( !empty($item[4]) )

{

$item[4] = str_replace( '"' , '' , $item[4]);



$result[ $item[2] ] = $item[4];

}

}

}

}



return $result;

}

import glob



out = open( 'tune1.h' , 'w' )

listOut = open( 'list.txt' , 'w' )



def writeTune(i, data):

out .write( '\nconst unsigned char tune_%s[] = {\n\t' % i)



t = 0



lst = []

for b in data:

lst.append(hex(ord(b)))

t+=1

if t % 16 == 0:

lst[-1] += '\n\t'



out .write( ', ' .join(lst))

out .write( '};\n\n' )



def main():

i = 0



for fn in glob.glob( '*.v2m' ):

f = open(fn, 'rb' )

data = f.read()



writeTune(i, data)

listOut.write( "%s, %s\n" % (i, fn))



i += 1



main()

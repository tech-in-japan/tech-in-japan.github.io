function stripXMLNamespaces(xml:XML):XML

{

var s:String = xml.toString();

var pattern1:RegExp = /\s*xmlns[^\'\ "]*=[\'\"][^\'\"]*[\'\"]/gi;

s = s.replace(pattern1, " ");

var pattern2:RegExp = /<[\/]{0,1}(\w+:).*?>/i;
